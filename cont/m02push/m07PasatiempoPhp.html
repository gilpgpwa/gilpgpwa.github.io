<?php
require_once __DIR__ .
 "/../mensajes/msgFaltaElNombre.php";
require_once __DIR__ .
 "/../mensajes/msgFaltaElId.php";

class Pasatiempo
{
 public $id;
 public $nombre;
 public function valida()
 {
  if ($this->nombre === null) {
   throw new Exception(
    msgFaltaElNombre()
   );
  }
 }
 public function validaCambio()
 {
  if ($this->id === null) {
   throw new Exception(
    msgFaltaElId()
   );
  }
  if ($this->nombre === null) {
   throw new Exception(
    msgFaltaElNombre()
   );
  }
 }
}
