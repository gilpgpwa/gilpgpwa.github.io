<?php
require_once __DIR__ .
 "/pasatiempoCrea.php";

class AccesoBd
{
 private static $con;

 public static
 function getCon(): \PDO
 {
  if (!self::$con) {
   self::$con = self::conecta();
   self::prepara();
  }
  return self::$con;
 }
 public static
 function conecta(): \PDO
 {
  return new PDO(
   "sqlite:phpbd.db",
   null,
   null,
   [PDO::ATTR_ERRMODE =>
   PDO::ERRMODE_EXCEPTION]
  );
 }
 public static function prepara()
 {
  pasatiempoCrea(self::getCon());
 }
}
