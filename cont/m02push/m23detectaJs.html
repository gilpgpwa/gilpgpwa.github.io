import {
 habilitaNotificaciones
} from "./habilitaNotificaciones.js"

/**
* @param {string | URL} sw
*/
export async function detecta(sw) {
 habilitaNotificaciones();
 if (!('PushManager' in window)) {
  throw new Error("Este " +
   "navegador no soporta " +
   "notificaciones push.");
 } else if ("serviceWorker"
  in navigator) {
  const registro = await navigator.
   serviceWorker.
   register(sw);
  console.log(sw, "registrado.");
  console.log(registro);
  if (!("showNotification"
   in registro)) {
   throw new Error("Este " +
    "navegador no soporta " +
    "notificaciones.");
  }
 } else {
  throw new Error("Este " +
   "navegador no soporta " +
   "service worker.");
 }
}
