import {
 execute
} from "../../lib/execute.js";
import {
 conn,
 PASATIEMPO
} from "./conn.js";

/** @param {any} modelo */
export async function
 pasatiempoModifica(modelo) {
 return execute(conn,
  [PASATIEMPO], tx => {
   modelo.modificacion = Date.now()
   const store =
    tx.objectStore(PASATIEMPO)
   store.put(modelo)
  })
}
