import {
 idbQuery
} from "../../lib/js/idbQuery.js"
import {
 conn,
 PASATIEMPO
} from "./conn.js"

/** @param {string} uuid */
export async function
 pasatiempoBusca(uuid) {
 return idbQuery(conn,
  [PASATIEMPO], (tx, resolve) => {
   const proc = tx.
    objectStore(PASATIEMPO).
    get(uuid)
   proc.onsuccess =
    () => resolve(proc.result)
  })
}
