import {
 idbQuery
} from "../../lib/js/idbQuery.js"
import {
 conn,
 PASATIEMPO
} from "./conn.js"

/**
 * Lista todos los objetos,
 * incluyendo los que tienen
 * borrado l√≥gico. */
export async function
 pasatiempoConsultaTodos() {
 return idbQuery(conn,
  [PASATIEMPO], (tx, resolve) => {
   const modelos = []
   const proc = tx.
    objectStore(PASATIEMPO).
    openCursor()
   proc.onsuccess = () => {
    const cursor = proc.result
    if (cursor === null) {
     resolve(modelos)
    } else {
     modelos.push(cursor.value)
     cursor.continue()
    }
   }
  })
}
