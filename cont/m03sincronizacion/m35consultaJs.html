/**
 * @param {Promise<IDBDatabase>} bd
 * @param {string[]} entidades
 * @param { (
 *  tx:
 *   IDBTransaction,
 *  resolve:
 *   (value: any)=>void
 *  ) => void } funcion
 */
export async function consulta(bd,
 entidades, funcion) {
 const con = await bd
 return new Promise(
  (resolve, reject) => {
   /* Inicia una transacción de
    * solo lectura. */
   const tx = con.transaction(
    entidades, "readonly")
   /* Al terminar con error hace
    * reject. */
   tx.onerror =
    () => reject(tx.error)
   /* Estas son las operaciones de
    * la transacción. */
   funcion(tx, resolve)
  })
}
