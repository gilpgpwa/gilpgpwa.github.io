import {
 muestraError
} from "../lib/js/muestraError.js"
import "./config.js"
import {
 pasatiempoAgrega
} from "./dao/pasatiempoAgrega.js"

const form =
 document.querySelector("form")
let uuid = 0

if (form !== null) {
 form.addEventListener("submit",
  guarda)
}

/** @param {Event} evt */
async function guarda(evt) {
 try {
  evt.preventDefault()
  if (form != null) {
   const nombre =
    form.nombre.value.trim()
   const modelo = {
    uuid: Date.now().toString()
     + Math.random() + uuid++,
    nombre
   };
   await pasatiempoAgrega(modelo)
   location.href = "index.html"
  }
 } catch (e) {
  muestraError(e)
 }
}
