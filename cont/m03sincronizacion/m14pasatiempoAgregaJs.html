import {
 idbExecute
} from "../../lib/js/idbExecute.js"
import {
 conn,
 PASATIEMPO
} from "./conn.js"

/** @param {any} modelo */
export async function
 pasatiempoAgrega(modelo) {
 return idbExecute(conn,
  [PASATIEMPO], tx => {
   modelo.modificacion = Date.now()
   modelo.eliminado = 0
   const store =
    tx.objectStore(PASATIEMPO)
   store.add(modelo)
  })
}
