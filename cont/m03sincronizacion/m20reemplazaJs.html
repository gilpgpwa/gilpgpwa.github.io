import {
 cambio
} from "../../lib/js/cambio.js"
import {
 conn,
 PASATIEMPO
} from "./conn.js"

/**
 * Borra el contenido del almac√©n y
 * guarda el contenido del listado.
 * @param {Array} lista nuevos
 *   datos. */
export async function
 reemplaza(lista) {
 return cambio(conn,
  [PASATIEMPO], tx => {
   const store =
    tx.objectStore(PASATIEMPO)
   store.clear()
   for (const modelo of lista) {
    store.add(modelo)
   }
  })
}
