import {
 idbQuery
} from "../../lib/js/idbQuery.js"
import {
 conn,
 PASATIEMPO
} from "./conn.js"

/**
 * Lista todos los objetos que no
 * tienen borrado l√≥gico.
 */
export async function
 pasatiempoConsulta() {
 return idbQuery(conn,
  [PASATIEMPO], (tx, resolve) => {
   const modelos = []
   const proc = tx.
    objectStore(PASATIEMPO).
    openCursor()
   proc.onsuccess = () => {
    const cursor = proc.result
    if (cursor === null) {
     resolve(modelos)
    } else {
     let modelo = cursor.value
     if (modelo.eliminado == 0) {
      modelos.push(modelo);
     }
     cursor.continue()
    }
   }
  })
}
