const cliente =
 require("./cliente")

/** @param {string} valor */
async function setValor(valor) {
 const client = cliente()
 try {
  await client.connect()
  const iot = client.db("iot")
  const dispositivo =
   iot.collection("dispositivo")
  const historico =
   iot.collection("historico")
  console.log("Env√≠a", "bd", valor)
  await dispositivo.updateOne(
   { _id: "1" },
   {
    $set: {
     valor
    }
   })
  /** @type {import("./tipos").
   *        historico} */
  const modelo = {
   dispositivoId: "1",
   valor,
   timestamp: Date.now()
  }
  await historico.insertOne(modelo)
 } catch (e) {
  console.log(e)
 } finally {
  client.close()
 }
}

module.exports = setValor
