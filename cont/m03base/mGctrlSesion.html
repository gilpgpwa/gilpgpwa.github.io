const CUE = "cue"
/** @type {HTMLFormElement|null} */
const sesion = document["sesion"]
/** @type {HTMLFormElement|null} */
const login = document["login"]

let esHumano = false

if (sesion !== null) {
 sesion.terminar.addEventListener(
  "click", terminaSesión)
}
if (login !== null) {
 login.addEventListener("submit",
  iniciaSesión)
 login.humano.addEventListener(
  "input", () => esHumano =
   login.humano.valueAsNumber > 0
 )
}

actualizaFormas()

function actualizaFormas() {
 const cue =
  sessionStorage.getItem(CUE)
 if (sesion !== null) {
  if (cue === null) {
   // Sesión terminada.
   sesion.hidden = true
   sesion.cue.value = ""
  } else {
   // Sesión iniciada.
   sesion.hidden = false
   sesion.cue.value = cue
  }
 }
 if (login !== null) {
  login.cue.value = ""
  login.match.value = ""
  login.humano.valueAsNumber = 0
  esHumano = false;
  login.hidden = cue !== null
 }
}
/** @param {Event} evt */
async function iniciaSesión(evt) {
 try {
  evt.preventDefault();
  if (login !== null
   && sesion !== null) {
   if (!esHumano) {
    throw new Error(
     "Comprueba que eres humano.")
   }
   // Toma los datos de la forma.
   const datos =
    new FormData(login)
   /* Usa el método POST para
    * enviar los datos de la forma
    * al recurso
    *  "/valida"
    * en el servidor. */
   const res = await fetch(
    "/valida",
    {
     method: "POST",
     body: datos
    })
   if (res.ok) {
    const resp = await res.text()
    if (resp === "ok") {
     sessionStorage.setItem(CUE,
      login.cue.value)
     actualizaFormas()
    } else {
     alert("Datos incorrectos.")
    }
   } else {
    throw new Error(res.statusText)
   }
  }
 } catch (e) {
  console.error(e)
  alert(e.message)
 }
}

async function terminaSesión() {
 sessionStorage.removeItem(CUE)
 actualizaFormas()
}
