I. getValor.js

const cliente =
 require("./cliente")

async function getValor() {
 const client = cliente()
 try {
  await client.connect()
  const iot = client.db("iot")
  const dispositivo =
   iot.collection("dispositivo")
  /** @type {import("./tipos").
   *       dispositivo|null} */
  const modelo = await dispositivo.
   findOne({ _id: "1" })
  console.log("Recibe", "bd",
   modelo)
  if (modelo !== null) {
   return modelo.valor
  } else {
   return "0"
  }
 } catch (e) {
  console.log(e)
 } finally {
  client.close()
 }
 return "0"
}

module.exports = getValor

J. setValor.js

const cliente =
 require("./cliente")

/** @param {string} valor */
async function setValor(valor) {
 const client = cliente()
 try {
  await client.connect()
  const iot = client.db("iot")
  const dispositivo =
   iot.collection("dispositivo")
  const historico =
   iot.collection("historico")
  console.log("Envía", "bd", valor)
  await dispositivo.updateOne(
   { _id: "1" },
   {
    $set: {
     valor
    }
   })
  /** @type {import("./tipos").
   *        historico} */
  const modelo = {
   dispositivoId: "1",
   valor,
   timestamp: Date.now()
  }
  await historico.insertOne(modelo)
 } catch (e) {
  console.log(e)
 } finally {
  client.close()
 }
}

module.exports = setValor

K. index.js

const express = require("express")
const { Server } = require('http')
const { join } = require("path")
const { server } =
 require('websocket')
const setValor =
 require("./setValor")
const getValor =
 require("./getValor")

const app = express()
const httpServer = new Server(app)
const www = join(__dirname, "www")

app.use(express.static(www))
const wsServer =
 new server({ httpServer })

getValor().
 then(valor => {
  const cons = new Set()
  wsServer.on('request', req => {
   const con = req.
    accept(undefined, req.origin)
   cons.add(con)
   console.log("Conexión.")
   console.log("Envía cliente",
    valor)
   con.sendUTF(valor)

   con.on("message", async men => {
    console.log("Recibe",
     // @ts-ignore
     "cliente", men.utf8Data)
    valor =
     valor === "0" ? "1" : "0"
    await setValor(valor)
    console.log("Envía",
     "clientes", valor)
    for (let item of cons.keys()) {
     item.sendUTF(valor)
    }
   })

   con.on("close",
    (razon, descripcion) => {
     cons.delete(con)
     console.log("Desconexión",
      razon, descripcion)
    })
  })
  httpServer.listen(3000,
   () => console.log(
    "Escuchando puerto 3000."))
 }).
 catch(e => console.log(e))
