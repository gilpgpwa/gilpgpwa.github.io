1. Instrucciones

  • Este ejercicio usa la librería WebPush para PHP. Puedes profundizar en este tema en la URL https://github.com/web-push-libs/web-push-php

  1. Revisa el proyecto en Replit con la URl https://replit.com/@GilbertoPacheco/phppush?v=1

  2. Usa o crea una cuenta de Google.

  3. Crea una cuenta de Replit usando la cuenta de Google.

  4. Crea un proyecto PHP Web Server en Replit y edita o sube los archivos de este proyecto.

  5. El ejercicio usa una llave pública en el servidor y la repite el código de JavaScript en el navegador. También se requiere la correspondiente llave privada que se almacena solo en el servidor.

  6. Las llaves se pueden generar con la URL https://phppush.gilbertopacheco.repl.co/srv/generallaves.php. Para copiar los valores despliega el código fuente de la página, porque en el navegador algunos caracteres pueden tener interpretación diferente.

  7. Para el contenido de la carpeta vendor y los archivos composer tienes 2 opciones:

    • Descargar el siguiente archivo zip: composer.zip.

    • Crear una carpeta, usar un shell como CMD, sh o PowerShell en esa carpeta, tener PHP y composer para PHP instalados en tu computadora y usar la orden:

      composer require minishlink/web-push

  8. Depura el proyecto.

  9. Crea la página de spotlight del proyecto.

class SrvSuscripcion
extends Servicio
{
 protected
 function implementacion()
 {
  $modelo = leeSuscripcion();
  $method =
   $_SERVER["REQUEST_METHOD"];

  if ($method === "POST") {
   suscripcionAgrega($modelo);
  } elseif ($method === "PUT") {
   $registrado = suscripcionBusca(
    $modelo->endpoint
   );

    suscripcionModifica($modelo);

   }
  } elseif ($method === "DELETE") {
   suscripcionElimina(
    $modelo->endpoint
   );
  }
  return [];
 }
}

export async function configura() {
 try {
  cambiaEstado(
   Estado.INCOMPATIBLE)
  if (btnSuscribe != null) {
   btnSuscribe.addEventListener(
    "click", suscribe)
  }
  if (btnCancela != null) {
   btnCancela.addEventListener(
    "click", cancela)
  }
  if (btnNotifica != null) {
   btnNotifica.addEventListener(
    "click", notifica)
  }
  await detecta("sw.js")
  await actualiza()
 } catch (e) {
  muestraError(e)
 }
}

async function actualiza() {
 const suscripcion =
  await getSuscripcion()
 if (suscripcion === null) {
  cambiaEstado(Estado.DESUSCRITO)
 } else {
  await srvSuscripcionProyx(
   suscripcion, Method.PUT)
  cambiaEstado(Estado.SUSCRITO)
 }
}

async function suscribe() {
 try {
  cambiaEstado(Estado.CALCULANDO)
  const registro =
   await navigator.serviceWorker.
    ready
  const suscripcion =
   await registro.pushManager.
    subscribe({
     userVisibleOnly: true,
     applicationServerKey
    })
  await srvSuscripcionProyx(
   suscripcion, Method.POST)
  cambiaEstado(Estado.SUSCRITO)
 } catch (e) {
  muestraError(e)
 }
}

async function cancela() {
 try {
  cambiaEstado(Estado.CALCULANDO)
  const suscripcion =
   await getSuscripcion()
  if (suscripcion !== null
   && await suscripcion.
    unsubscribe()) {
   await srvSuscripcionProyx(
    suscripcion, Method.DELETE)
   cambiaEstado(Estado.DESUSCRITO)
  }
 } catch (e) {
  muestraError(e)
 }
}

/** @param {Estado} estado */
function cambiaEstado(estado) {
 if (outEstado !== null
  && btnSuscribe !== null
  && btnNotifica !== null
  && btnCancela !== null) {
  outEstado.value = estado
  if (estado
   === Estado.INCOMPATIBLE ||
   estado === Estado.CALCULANDO) {
   btnSuscribe.hidden = true
   btnCancela.hidden = true
   btnNotifica.hidden = true
  } else if (estado
   === Estado.SUSCRITO) {
   btnSuscribe.hidden = true
   btnCancela.hidden = false
   btnNotifica.hidden = false
  } else if (estado
   === Estado.DESUSCRITO) {
   btnSuscribe.hidden = false
   btnCancela.hidden = true
   btnNotifica.hidden = true
  }
 }
}

 srvSuscripcionProyx(suscripcion,
  method) {
 const key =
  suscripcion.getKey("p256dh")
 const token =
  suscripcion.getKey("auth")
 const encodings = PushManager
  .supportedContentEncodings ||
  ["aesgcm"]
 await ejecuta(fetch(
  "srv/SrvSuscripcion.php", {
  method,
  body: JSON.stringify({
   endpoint: suscripcion.endpoint,
   publicKey: key
    ? btoa(
     String.fromCharCode.apply(
      null, new Uint8Array(key)))
    : null,
   authToken: token
    ? btoa(
     String.fromCharCode.apply(
      null,
      new Uint8Array(token)))
    : null,
   contentEncoding: encodings[0]
  })
 }))
}

async function notifica(evt) {

 if (data != null
  && self instanceof
  ServiceWorkerGlobalScope

 habilitaNotificaciones() {
 if ("Notification" in window) {
  let permiso =
   Notification.permission;
  if (permiso === "default") {
   permiso = await Notification.
    requestPermission()
  }
  if (permiso === "denied") {
   throw new Error(
    "Notificaciones bloqueadas.")

 } else {
  throw new Error("Este " +
   "navegador no soporta " +

export function
 urlBase64ToUint8Array(
  base64String) {
 const padding = '='.repeat(
  (4 - (base64String.length % 4))
  % 4);
 const base64 =
  (base64String + padding).
   replace(/\-/g, '+').
   replace(/_/g, '/')

 const rawData = atob(base64)
 const outputArray =
  new Uint8Array(rawData.length)

 for (let i = 0
  ; i < rawData.length; ++i) {
  outputArray[i] =
   rawData.charCodeAt(i)
 }
 return outputArray
}

class SrvSincro extends Servicio
{
 protected
 function implementacion()
 {
  $lista = leeJson();
  if (!$lista) {
   $lista = [];
  }
  foreach ($lista as $modCli) {
   $modServ =
    pasatiempoBusca($modCli->uuid);
   if ($modServ) {
    if (
     !$modServ->eliminado
     && $modCli->eliminado
    ) {
     /* CONFLICTO. El registro está
      * en el servidor, donde no se
      * ha eliminado, pero ha sido
      * eliminado en el cliente.
      * Gana el cliente, porque
      * optamos por no revivir lo
      * que se ha borrado. */
     pasatiempoModifica($modCli);
    } else if (
     !$modCli->eliminado
     && !$modServ->eliminado
    ) {
     /* CONFLICTO. El registro está
      * tanto en el servidor como
      * en el cliente. Los datos
      * pueden ser diferentes.
      * PREVALECE LA FECHA MÁS
      * GRANDE. Cuando gana el
      * servidor no se hace nada.*/
     if (
      $modCli->modificacion
      > $modServ->modificacion
     ) {
      /* La versión del cliente es
       * más nueva y prevalece. */
      pasatiempoModifica($modCli);
     }
    }
   } else {
    /* CONFLICTO. El objeto no ha
     * estado en el servidor.
     * AGREGARLO solamente si no
     * está eliminado. */
    if (!$modCli->eliminado) {
     pasatiempoAgrega($modCli);
    }
   }
  }
  $lista = pasatiempoConsulta();
  return ["lista" => $lista];
 }
}

 "Pasatiempo"

/** @type {Promise<IDBDatabase>} */
export const conn =
 new Promise((resolve, reject) => {
  const solicitud = indexedDB.open(
   BD_NOMBRE, BD_VERSION)

   () => reject(solicitud.error)
  solicitud.onsuccess =
   () => resolve(solicitud.result)
  /* Automáticamente se inicia una
   * transacción para cambio de

     * existe. */
    if (bd.objectStoreNames
     .contains(PASATIEMPO)) {
     bd.deleteObjectStore(
      PASATIEMPO)
    }
    /* Crea el almacén "Pasatiempo"
     * con el campo llave "uuid".*/
    bd.
     createObjectStore(PASATIEMPO,
      { keyPath: "uuid" })
   }
 });

 console.log(
  "Intentando cargar cache:",
  CACHE)
 // Borra todos los chaches.
 const keys = await caches.keys()
 for (const key of keys) {
  await caches.delete(key)
 }

 const cache =
  await caches.open(CACHE)

 // Busca el contenido del cache.
 const cache =
  await caches.open(CACHE)

  await cache.match(evt.request,
   { ignoreSearch: true })

  return response

/**
 * @param {Promise<IDBDatabase>} bd
 * @param {string[]} entidades
 * @param {(
 *   tx: IDBTransaction) => void
 *  } funcion
 */
export async function idbExecute(bd,
 entidades, funcion) {
 const con = await bd
 return new Promise(
  (resolve, reject) => {
   /* Inicia una transacción de
    * lectura y escritura. */
   const tx = con.transaction(
    entidades, "readwrite")
   /* Al terminar con éxito hace
    * resolve. */
   tx.oncomplete = resolve;
   /* Al terminar con error hace
    * reject. */
   tx.onerror =
    () => reject(tx.error)
   /* Estas son las operaciones de
    * la transacción. */
   funcion(tx)
  })
}

/** Codifica un texto para que

  toString().

/** @param {string} letra */
function reemplaza(letra) {
 switch (letra) {
  case "<": return "&lt;"
  case ">": return "&gt;"
  case '"': return "&quot;"
  case "'": return "&#039;"
  default: return letra
 }
}

 } else if (
